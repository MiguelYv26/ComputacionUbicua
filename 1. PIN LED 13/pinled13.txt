void setup () {
  pinMode(13, OUTPUT);		// pin 13 como salida.
}							

void loop(){
  digitalWrite(13, HIGH);	// pin 13 a nivel o valor alto (1).
  delay(1000);				// demora de 1 seg (1000 ms).
  digitalWrite(13, LOW);	// pin 13 a nivel o valor bajo (0).
  delay(1000);				// delay de 1 seg (1000 ms).
}
